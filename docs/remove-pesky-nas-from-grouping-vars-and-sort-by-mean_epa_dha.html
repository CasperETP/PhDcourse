Chapter 5 Remove pesky NAs from grouping vars and sort by mean_EPA_DHA

penguins %>% 
  filter(!is.na(sex)) %>% 
  group_by(species, sex) %>%
  summarize(
    mean_bm =mean(body_mass_g, na.rm=T),
    sd_bm   =sd(body_mass_g, na.rm=T),
    n            = n()
)%>% 
arrange(desc(mean_bm))
